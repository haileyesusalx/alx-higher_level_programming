#!/usr/bin/python3
"""
Test cases for BaseGeometry class
"""

import unittest
from models.base_geometry import BaseGeometry


class TestBaseGeometry(unittest.TestCase):
    """
    TestBaseGeometry class
    """

    def test_area(self):
        """
        Test area method
        """
        bg = BaseGeometry()
        with self.assertRaises(Exception) as e:
            bg.area()
        self.assertEqual(str(e.exception), "area() is not implemented")

    def test_integer_validator(self):
        """
        Test integer_validator method
        """
        bg = BaseGeometry()
        self.assertRaises(TypeError, bg.integer_validator, "name", "not_int")
        self.assertRaises(ValueError, bg.integer_validator, "name", 0)
        self.assertRaises(ValueError, bg.integer_validator, "name", -1)


if __name__ == '__main__':
    unittest.main()
